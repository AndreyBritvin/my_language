BITS 64
section .data
    fmt db "Hello world", 10, 0

section .text
global main
extern out
extern hlt


main:
push rbp
mov rbp, rsp
sub rsp, 8
push 10
pop qword [rbp - 8]               ; first
;put in this ^ value: ..........

push qword [rbp - 8]              ; first
push 3
pop rcx
pop rbx
cmp rbx, rcx
jbe IF_LABEL_0
push 2
call out
;print ..

IF_LABEL_0:

push qword [rbp - 8]              ; first
push 2
pop rcx
pop rbx
cmp rbx, rcx
jae IF_LABEL_2
push 1
call out
;print .

IF_LABEL_2:
call out

call hlt
mov rsp, rbp
pop rbp
ret
